        // ==================== ESTADO GLOBAL ====================
        let currentRole = 0;
        const roles = ['Sin sesión', 'Cliente', 'Asesor Jurídico', 'Administrador'];
        let currentView = 0;

        const chatState = {
            pub: { history: [], count: 0, limit: 3, typing: false },
            client: { history: [], count: 0, limit: 999, typing: false }
        };

        // ==================== NAVEGACIÓN ====================
        function switchView(n) {
            currentView = n;
            document.querySelectorAll('.view').forEach((v, i) => {
                v.classList.toggle('active', i === n);
            });
            document.querySelectorAll('.tab').forEach((t, i) => {
                t.classList.toggle('active', i === n);
            });
        }

        function cycleRole() {
            currentRole = (currentRole + 1) % roles.length;
            document.getElementById('rolePill').textContent = roles[currentRole];
            const sv = currentRole === 0 ? 0 : currentRole;
            switchView(sv);
            toast(`🔄 Perfil cambiado a: ${roles[currentRole]}`);
        }

        // ==================== TOAST ====================
        function toast(msg) {
            const t = document.getElementById('toast');
            t.textContent = msg;
            t.classList.add('show');
            clearTimeout(t._timeout);
            t._timeout = setTimeout(() => t.classList.remove('show'), 2800);
        }

        // ==================== MODAL ====================
        function openEvModal(title, subtitle) {
            document.getElementById('modalTitle').textContent = title;
            document.getElementById('modalSubtitle').textContent = subtitle;
            document.getElementById('evidenceModal').classList.add('show');
        }

        function closeEvModal(e) {
            if (e && e.target !== document.getElementById('evidenceModal')) return;
            document.getElementById('evidenceModal').classList.remove('show');
        }

        function fakeUpload() {
            closeEvModal();
            toast('✅ Archivo guardado con sello de tiempo y GPS');
        }

        // ==================== FOTO AVATAR ====================
        function previewPhoto(input, previewId, placeholderId) {
            const file = input.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = function(e) {
                const preview = document.getElementById(previewId);
                const placeholder = document.getElementById(placeholderId);
                preview.src = e.target.result;
                preview.style.display = 'block';
                placeholder.style.display = 'none';
                toast('📷 Foto actualizada correctamente');
            };
            reader.readAsDataURL(file);
        }

        // ==================== CHAT ====================
        function autoResize(el) {
            el.style.height = 'auto';
            el.style.height = Math.min(el.scrollHeight, 120) + 'px';
        }

        function handleKey(e, chatId) {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                sendMsg(chatId);
            }
        }

        async function sendMsg(chatId) {
            const input = document.getElementById(chatId === 'pub' ? 'pubInput' : 'clientInput');
            const text = input.value.trim();
            if (!text || chatState[chatId].typing) return;

            input.value = '';
            input.style.height = 'auto';

            const state = chatState[chatId];
            state.count++;
            state.history.push({ role: 'user', content: text });

            // Agregar mensaje usuario al DOM
            const msgsContainer = document.getElementById(chatId === 'pub' ? 'pubMessages' : 'clientMessages');
            addMessage(msgsContainer, 'user', text);

            // Verificar límite para chat público
            if (chatId === 'pub' && state.count > state.limit) {
                showUpsell(msgsContainer);
                return;
            }

            // Mostrar typing
            showTyping(msgsContainer);
            state.typing = true;

            try {
                const systemPrompt = chatId === 'pub' 
                    ? getPublicSystemPrompt() 
                    : getClientSystemPrompt();

                const response = await fetch('https://api.anthropic.com/v1/messages', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        model: 'claude-sonnet-4-20250514',
                        max_tokens: 1000,
                        system: systemPrompt,
                        messages: state.history
                    })
                });

                const data = await response.json();
                removeTyping(msgsContainer);
                state.typing = false;

                const replyText = data.content[0].text;
                state.history.push({ role: 'assistant', content: replyText });
                addMessage(msgsContainer, 'bot', replyText);

            } catch (error) {
                removeTyping(msgsContainer);
                state.typing = false;
                addMessage(msgsContainer, 'bot', 'Lo siento, tuve un problema técnico. Por favor intenta de nuevo en unos momentos.');
            }
        }

ESPECIALIDADES: horas extra, liquidación, finiquito, vacaciones, aguinaldo, IMSS, prima dominical, riesgo de trabajo en trayecto, cambio de domicilio.

        function formatReply(text) {
            return text
                .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
                .replace(/\n\n/g, '<br><br>')
                .replace(/\n/g, '<br>');
        }
